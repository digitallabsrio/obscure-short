package main
import (
  "fmt"
  "sync"
  "time"
)
func goroutine1(wg *sync.WaitGroup) {
  time.Sleep(time.Second)
  fmt.Println("goroutine1")
  wg.Done()
}
func goroutine2(wg *sync.WaitGroup) {
  time.Sleep(time.Second)
  fmt.Println("goroutine2")
  wg.Done()
}
func main() {
  wg := &sync.WaitGroup{}
  wg.Add(2)
  go goroutine1(wg)
  go goroutine2(wg)
  wg.Wait()
}
